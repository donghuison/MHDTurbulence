139void HLLD(const double (&leftst)[2*mconsv+madd],const double (&rigtst)[2*mconsv+madd],double (&nflux)[mconsv]){

142//
143// HLLD Scheme
144//
145// Purpose
146// Calculation of Numerical Flux by HLLD method
147//
148// Reference
149//
150// Input
151// Output
152//=====================================================================
153
154//----- U -----
155// qql :: left state
156// qqr :: right state
157 double rol,vxl,vyl,vzl,ptl,eel;
158 double ror,vxr,vyr,vzr,ptr,eer;
159 double rxl,ryl,rzl;
160 double rxr,ryr,rzr;
161 double bxs,byl,bzl;
162 double byr,bzr;
163 double ptst;
164 double scl[ncomp],scr[ncomp];
165//----- U* ----
166// qqlst :: left state
167// qqrst :: right state
168 double rolst,vxlst,vylst,vzlst,eelst;
169 double rorst,vxrst,vyrst,vzrst,eerst;
170 double rxlst,rylst,rzlst;
171 double rxrst,ryrst,rzrst;
172 double bylst,bzlst;
173 double byrst,bzrst;
174 double sclst[ncomp],scrst[ncomp];
175
176//----- U** ----
177// qqlst :: left state
178// qqrst :: right state
179 double vyldst,vzldst,eeldst;
180 double vyrdst,vzrdst,eerdst;
181 double ryldst,rzldst;
182 double ryrdst,rzrdst;
183 double byldst,bzldst;
184 double byrdst,bzrdst;
185
186//----- flux ---
187// fqql :: left physical flux
188// fqqr :: right physical flux
189 double frol,frxl,fryl,frzl,feel;
190 double fbyl,fbzl;
191 double fror,frxr,fryr,frzr,feer;
192 double fbyr,fbzr;
193 double fscl[ncomp],fscr[ncomp];
194
195//----- wave speed ---
196// sl :: left-going fastest signal velocity
197// sr :: right-going fastest signal velocity
198// sm :: contact discontinuity velocity
199// slst :: left-going alfven velocity
200// srst :: right-going alfven velocity
201 double sm,sl,sr,slst,srst;
202
203// cfl :: left-state Fast wave velocity
204// cfr :: right-sate Fast wave velocity
205 double cfl,cfr;
206
207//--------------------
208// temporary variables
209 double sdl,sdr,sdml,sdmr,isdml,isdmr,rosdl,rosdr;
210 double temp;
211
212// no if
213 double sign1,maxs1,mins1;
214 double msl,msr,mslst,msrst,temp1,invsumro,sqrtror,sqrtrol,abbx;
215 double bxsq,temp_fst,eps,itf,vdbstl,vdbstr,signbx;
216
217//----- Step 0. ----------------------------------------------------------|
218 eps = 1.0e-30;
219//---- Left state
220
221 rol = leftst[mudn];
222 eel = leftst[muet];
223 rxl = leftst[muvu];
224 ryl = leftst[muvv];
225 rzl = leftst[muvw];
226 vxl = leftst[muvu]/leftst[mudn];
227 vyl = leftst[muvv]/leftst[mudn];
228 vzl = leftst[muvw]/leftst[mudn];
229 byl = leftst[mubv];
230 bzl = leftst[mubw];
231 ptl = leftst[mpre];
232 for(int n=0;n<ncomp;n++){
233 scl[n] = leftst[mst+n];
234 }
235//---- Right state
236
237 ror = rigtst[mudn];
238 eer = rigtst[muet];
239 rxr = rigtst[muvu];
240 ryr = rigtst[muvv];
241 rzr = rigtst[muvw];
242 vxr = rigtst[muvu]/rigtst[mudn];
243 vyr = rigtst[muvv]/rigtst[mudn];
244 vzr = rigtst[muvw]/rigtst[mudn];
245 byr = rigtst[mubv];
246 bzr = rigtst[mubw];
247 ptr = rigtst[mpre];
248 for(int n=0;n<ncomp;n++){
249 scr[n] = rigtst[mst+n];
250 }
251//----- Step 1. ----------------------------------------------------------|
252// Compute wave left & right wave speed
253//
254 cfl = leftst[mcsp];
255 cfr = rigtst[mcsp];
256
257 sl = std::min(vxl,vxr)-std::max(cfl,cfr); // note sl is negative
258 sr = std::max(vxl,vxr)+std::max(cfl,cfr);
259//----- Step 2. ----------------------------------------------------------|
260// compute L/R fluxs
261//
262// Left value
263 frol = leftst[mfdn];
264 feel = leftst[mfet];
265 frxl = leftst[mfvu];
266 fryl = leftst[mfvv];
267 frzl = leftst[mfvw];
268 fbyl = leftst[mfbv];
269 fbzl = leftst[mfbw];
270 for(int n=0;n<ncomp;n++){
271 fscl[n] = leftst[mfst+n];
272 }
273
274// Right value
275 fror = rigtst[mfdn];
276 feer = rigtst[mfet];
277 frxr = rigtst[mfvu];
278 fryr = rigtst[mfvv];
279 frzr = rigtst[mfvw];
280 fbyr = rigtst[mfbv];
281 fbzr = rigtst[mfbw];
282
283 for(int n=0;n<ncomp;n++){
284 fscr[n] = rigtst[mfst+n];
285 }
286
287//----- Step 4. ----------------------------------------------------------|
288// compute middle and alfven wave
289//
290 sdl = std::min(-1e-20,sl - vxl);
291 sdr = std::max( 1e-20,sr - vxr);
292 rosdl = rol*sdl;
293 rosdr = ror*sdr;
294
295 temp = 1.0e0/(rosdr - rosdl);
296// Eq. 45
297 sm = (rosdr*vxr - rosdl*vxl - ptr + ptl)*temp;
298
299 sdml = std::min(-1e-20,sl - sm); isdml = 1.0e0/sdml;
300 sdmr = std::max( 1e-20,sr - sm); isdmr = 1.0e0/sdmr;
301
302//----- Step 5. ----------------------------------------------------------|
303// compute intermediate states
304//
305// Eq. 49
306 ptst = (rosdr*ptl-rosdl*ptr+rosdl*rosdr*(vxr-vxl))*temp;
307
308//----- Step 5A. ----------------------------------------------------------|
309// compute Ul*
310//
311 bxs = 0.5e0*(leftst[mubu]+rigtst[mubu]);
312 bxsq = bxs*bxs;
313 temp_fst = rosdl*sdml - bxsq;
314 sign1 = std::copysign(1.0e0,std::abs(temp_fst)-eps);
315
316 maxs1 = std::max(0.0e0,sign1);
317 mins1 = std::min(0.0e0,sign1);
318
319 itf = 1.0e0/(temp_fst+mins1);
320 isdml = 1.0e0/sdml;
321
322 temp = bxs*(sdl-sdml)*itf;
323 rolst = maxs1*(rosdl*isdml) - mins1*rol;
324 for(int n=0;n<ncomp;n++){
325 sclst[n] = rolst/rol*scl[n];
326 }
327 vxlst = maxs1*sm - mins1*vxl;
328 rxlst = rolst*vxlst;
329
330 vylst = maxs1*(vyl-byl*temp) - mins1*vyl;
331 rylst = rolst*vylst;
332 vzlst = maxs1*(vzl-bzl*temp) - mins1*vzl;
333 rzlst = rolst*vzlst;
334
335 temp = (rosdl*sdl-bxsq)*itf;
336 bylst = maxs1*(byl*temp) - mins1*byl;
337 bzlst = maxs1*(bzl*temp) - mins1*bzl;
338
339 vdbstl = vxlst*bxs+vylst*bylst+vzlst*bzlst;
340 eelst = maxs1*(sdl*eel - ptl*vxl + ptst*sm +
341 bxs*(vxl*bxs+vyl*byl+vzl*bzl-vdbstl))*isdml
342 - mins1*eel ;
343
344//----- Step 5B. ----------------------------------------------------------|
345// compute Ur*
346//
347 temp_fst = rosdr*sdmr - bxsq;
348 sign1 = std::copysign(1.0e0,std::abs(temp_fst)-eps);
349 maxs1 = std::max(0.0e0,sign1);
350 mins1 = std::min(0.0e0,sign1);
351
352 itf = 1.0e0/(temp_fst+mins1);
353 isdmr = 1.0e0/sdmr;
354
355 temp = bxs*(sdr-sdmr)*itf;
356 rorst = maxs1*(rosdr*isdmr) - mins1*ror;
357 for(int n=0;n<ncomp;n++){
358 scrst[n] = rorst/ror*scr[n];
359 }
360 vxrst = maxs1*sm - mins1*vxr;
361 rxrst = rorst*vxrst;
362
363 vyrst = maxs1*(vyr-byr*temp) - mins1*vyr;
364 ryrst = rorst*vyrst;
365 vzrst = maxs1*(vzr-bzr*temp) - mins1*vzr;
366 rzrst = rorst*vzrst;
367
368 temp = (rosdr*sdr-bxsq)*itf;
369 byrst = maxs1*(byr*temp) - mins1*byr;
370 bzrst = maxs1*(bzr*temp) - mins1*bzr;
371
372 vdbstr = vxrst*bxs+vyrst*byrst+vzrst*bzrst;
373 eerst = maxs1*((sdr*eer - ptr*vxr + ptst*sm)*isdmr +
374 bxs*(vxr*bxs+vyr*byr+vzr*bzr-vdbstr)*isdmr)
375 - mins1*eer;
376;
377//----- Step 5C. ----------------------------------------------------------|;
378// compute Ul** and Ur**;
379//;
380 sqrtrol = sqrt(rolst);
381 sqrtror = sqrt(rorst);
382
383 abbx = std::abs(bxs);
384 signbx = std::copysign(1.0e0,bxs);
385 sign1 = std::copysign(1.0e0,abbx-eps);
386
387 maxs1 = std::max(0e0,sign1);
388 mins1 = -std::min(0e0,sign1);
389 invsumro = maxs1/(sqrtrol + sqrtror);
390
391 temp = invsumro*(sqrtrol*vylst + sqrtror*vyrst
392 + signbx*(byrst-bylst));
393 vyldst = vylst*mins1 + temp;
394 vyrdst = vyrst*mins1 + temp;
395 ryldst = rylst*mins1 + rolst * temp;
396 ryrdst = ryrst*mins1 + rorst * temp;
397
398 temp = invsumro*(sqrtrol*vzlst + sqrtror*vzrst
399 + signbx*(bzrst-bzlst));
400 vzldst = vzlst*mins1 + temp;
401 vzrdst = vzrst*mins1 + temp;
402 rzldst = rzlst*mins1 + rolst * temp;
403 rzrdst = rzrst*mins1 + rorst * temp;
404
405 temp = invsumro*(sqrtrol*byrst + sqrtror*bylst
406 + signbx*sqrtrol*sqrtror*(vyrst-vylst));
407 byldst = bylst*mins1 + temp;
408 byrdst = byrst*mins1 + temp;
409
410 temp = invsumro*(sqrtrol*bzrst + sqrtror*bzlst
411 + signbx*sqrtrol*sqrtror*(vzrst-vzlst));
412 bzldst = bzlst*mins1 + temp;
413 bzrdst = bzrst*mins1 + temp;
414
415 temp = sm*bxs + vyldst*byldst + vzldst*bzldst;
416 eeldst = eelst - sqrtrol*signbx*(vdbstl - temp)*maxs1;
417 eerdst = eerst + sqrtror*signbx*(vdbstr - temp)*maxs1;
418
419//----- Step 6. ----------------------------------------------------------|;
420// compute flux;
421 slst = (sm - abbx/sqrtrol)*maxs1;
422 srst = (sm + abbx/sqrtror)*maxs1;
423
424 sign1 = std::copysign(1.0e0,sm);
425 maxs1 = std::max(0.0e0,sign1);
426 mins1 = -std::min(0.0e0,sign1);
427
428 msl = std::min(sl,0.0e0);
429 msr = std::max(sr,0.0e0);
430 mslst = std::min(slst,0.0e0);
431 msrst = std::max(srst,0.0e0);
432
433 temp = mslst-msl;
434 temp1 = msrst-msr;
435
436 nflux[mden] = (frol+(rolst-rol)*msl)*maxs1
437 +(fror+(rorst-ror)*msr)*mins1;
438 nflux[meto] = (feel+(eelst-eel)*msl+(eeldst-eelst)*mslst)*maxs1
439 +(feer+(eerst-eer)*msr+(eerdst-eerst)*msrst)*mins1;
440 nflux[mrvu] = (frxl+(rxlst-rxl)*msl)*maxs1
441 +(frxr+(rxrst-rxr)*msr)*mins1;
442 nflux[mrvv] = (fryl+(rylst-ryl)*msl+(ryldst-rylst)*mslst)*maxs1
443 +(fryr+(ryrst-ryr)*msr+(ryrdst-ryrst)*msrst)*mins1;
444 nflux[mrvw] = (frzl+(rzlst-rzl)*msl+(rzldst-rzlst)*mslst)*maxs1
445 +(frzr+(rzrst-rzr)*msr+(rzrdst-rzrst)*msrst)*mins1;
446 nflux[mbmu] = 0e0;
447 nflux[mbmv] = (fbyl+(bylst-byl)*msl+(byldst-bylst)*mslst)*maxs1
448 +(fbyr+(byrst-byr)*msr+(byrdst-byrst)*msrst)*mins1;
449 nflux[mbmw] = (fbzl+(bzlst-bzl)*msl+(bzldst-bzlst)*mslst)*maxs1
450 +(fbzr+(bzrst-bzr)*msr+(bzrdst-bzrst)*msrst)*mins1;
451 for (int n=0;n<ncomp;n++){
452 nflux[mst+n] = (fscl[n]+(sclst[n]-scl[n])*msl)*maxs1
453 +(fscr[n]+(scrst[n]-scr[n])*msr)*mins1;
454 }
