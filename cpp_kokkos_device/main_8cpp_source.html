126static void GenerateProblem(hydflux_mod::GridArray<double>& G,hydflux_mod::FieldArray<double>& P,hydflux_mod::FieldArray<double>& U) {
127 // Kelvin-Helmholtz initial condition
128 using namespace resolution_mod;
129 using namespace mpi_config_mod;
130 using namespace hydflux_mod;
131
132 const double pi = std::acos(-1.0);
133
134 // Parameters (same as Fortran)
135 const double rho1 = 1.0;
136 const double rho2 = 1.0;
137 const double dv = 2.0;
138 const double wid = 0.05;
139 const double sig = 0.2;
140 const double B0 = std::sqrt(2.0/3.0);
141 (void)rho1; (void)rho2; (void)B0; // currently symmetric / B0 unused in reference file
142
143 // Random perturbation strength (Fortran rrv)
144 const double rrv = 0.0e-2;
145
146 // Isothermal: choose csiso so that p = rho*csiso^2 matches Fortran p=1 when rho=1
147 //csiso = 1.0;
148 //#pragma omp target update to (csiso)
149
150 chg = 0.0;
151
152 // Base profile (fill including ghost cells to avoid uninitialized values)
153 for (int k = ks-ngh; k <= ke+ngh; ++k)
154 for (int j = js-ngh; j <= je+ngh; ++j)
155 for (int i = is-ngh; i <= ie+ngh; ++i) {
156 const double x = G.x1b(i);
157 const double y = G.x2b(j);
158
159 const double tanh_p = std::tanh((y + 0.5)/wid);
160 const double tanh_m = std::tanh((y - 0.5)/wid);
161
162 P(nden,k,j,i) = 1.0;
163 P(npre,k,j,i) = 1.0;
164
165 P(nve1,k,j,i) = 0.5 * dv * (tanh_p - tanh_m - 1.0);
166 P(nve2,k,j,i) = 1.0e-2 * std::sin(2.0*pi*x) *
167 ( std::exp(- (y + 0.5)*(y + 0.5)/(sig*sig))
168 + std::exp(- (y - 0.5)*(y - 0.5)/(sig*sig)) );
169 P(nve3,k,j,i) = 0.0;
170
171 P(nbm1,k,j,i) = 0.0;
172 P(nbm2,k,j,i) = 0.0;
173 P(nbm3,k,j,i) = 0.0;
174 P(nbps,k,j,i) = 0.0;
175
176 // specific internal energy is not used in the isothermal closure, but keep it consistent
177 P(nene,k,j,i) = P(npre,k,j,i) /(gam-1.0);
178 P(ncsp,k,j,i) = sqrt(gam*P(npre,k,j,i)/P(nden,k,j,i));
179
180 //P(ncsp,k,j,i) = csiso;
181 // Composition (Fortran: Xcomp(1)=0.5*(tanh((y+0.5)/wid)-tanh((y-0.5)/wid)))
182 for (int n=0; n<ncomp; ++n) {
183 (void)n; // for future multi-comp extension
184 P(nst, k,j,i) = 0.5 * (tanh_p - tanh_m);
185 }
186 }
187
188 // Random perturbation on v1 (Fortran: one random number per (j,k), applied to all i)
189 // Fortran uses random_seed with seed(1)=1, seed(2)=1 + myid_w*in*jn*kn.
190 // Here we emulate with a deterministic per-rank seed.
191 std::mt19937_64 rng(static_cast<unsigned long long>(1 + myid_w * 1000003ULL));
192 std::uniform_real_distribution<double> uni(0.0, 1.0);
193
194 for (int k = ks; k <= ke; k++)
195 for (int j = js; j <= je; j++) {
196 // const double r = uni(rng);
197 // const double dv_rand = dv * rrv * (r - 0.5);
198 for (int i = is; i <= ie; ++i) {
199 //P(nve1,k,j,i) += dv_rand;
200 }
201 }
202
203 namespace res = resolution_mod;
204 for (int k = ks; k <= ke; k++)
205 for (int j = js; j <= je; j++) {
206 const double y = G.x2b(j);
207 const double z = G.x3b(k);
208 const double dv_harm = dv * rrv * std::cos(6*2*pi*(y - res::x2min)/(res::x2max - res::x2min))
209 * std::cos(6*2*pi*(z - res::x3min)/(res::x3max - res::x3min));
210 for (int i = is; i <= ie; i++) {
211 P(nve1,k,j,i) += dv_harm;
212 }
213 }
214
215
216 // Build conserved variables U from primitives P
217 for (int k = ks-ngh; k <= ke+ngh; ++k)
218 for (int j = js-ngh; j <= je+ngh; ++j)
219 for (int i = is-ngh; i <= ie+ngh; ++i) {
220 const double rho = P(nden,k,j,i);
221 U(mden,k,j,i) = rho;
222 U(mrv1,k,j,i) = rho * P(nve1,k,j,i);
223 U(mrv2,k,j,i) = rho * P(nve2,k,j,i);
224 U(mrv3,k,j,i) = rho * P(nve3,k,j,i);
225
226 const double ekin = 0.5 * rho * ( P(nve1,k,j,i)*P(nve1,k,j,i)
227 + P(nve2,k,j,i)*P(nve2,k,j,i)
228 + P(nve3,k,j,i)*P(nve3,k,j,i) );
229 const double emag = 0.5 * ( P(nbm1,k,j,i)*P(nbm1,k,j,i)
230 + P(nbm2,k,j,i)*P(nbm2,k,j,i)
231 + P(nbm3,k,j,i)*P(nbm3,k,j,i) );
232
233 U(meto,k,j,i) = rho * P(nene,k,j,i) + ekin + emag;
234
235 U(mbm1,k,j,i) = P(nbm1,k,j,i);
236 U(mbm2,k,j,i) = P(nbm2,k,j,i);
237 U(mbm3,k,j,i) = P(nbm3,k,j,i);
238 U(mbps,k,j,i) = P(nbps,k,j,i);
239
240 for (int n=0; n<ncomp; ++n) {
241 U(mst+n,k,j,i) = rho * P(nst+n,k,j,i);
242 }
243 }
244}
