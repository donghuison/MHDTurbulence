40void InitializeMPI() {
41 MPI_Init(nullptr, nullptr);
42 MPI_Comm_size(MPI_COMM_WORLD, &nprocs_w);
43 MPI_Comm_rank(MPI_COMM_WORLD, &myid_w);
44
45 if (myid_w == 0) {
46 std::printf("MPI process= %d\n", nprocs_w);
47 std::printf("decomposition= %d %d %d\n", ntiles[0], ntiles[1], ntiles[2]);
48 }
49 MPI_Bcast( ntiles, ndim, MPI_INT, 0, MPI_COMM_WORLD);
50 MPI_Bcast(periodic, ndim, MPI_INT, 0, MPI_COMM_WORLD);
51
52 // Split communicator into groups of size ntiles[0]*ntiles[1]*ntiles[2]
53 int np_hyd = ntiles[0] * ntiles[1] * ntiles[2];
54 int color = (np_hyd > 0) ? (myid_w / np_hyd) : 0;
55 int key = myid_w;
56 MPI_Comm_split(MPI_COMM_WORLD, color, key, &mpi_comm_hyd);
57 MPI_Comm_size(mpi_comm_hyd, &nprocs_hyd);
58 MPI_Comm_rank(mpi_comm_hyd, &myid_hyd);
59
60 // Create 3D Cartesian topology
61 MPI_Cart_create(mpi_comm_hyd, ndim, ntiles, periodic, reorder, &comm3d);
62 MPI_Comm_rank(comm3d, &myid);
63 MPI_Comm_size(comm3d, &nprocs);
64
65 // Neighbors and my coordinates
66 MPI_Cart_shift(comm3d, dir1, 1, &n1m, &n1p);
67 MPI_Cart_shift(comm3d, dir2, 1, &n2m, &n2p);
68 MPI_Cart_shift(comm3d, dir3, 1, &n3m, &n3p);
69 MPI_Cart_coords(comm3d, myid, 3,coords);
70
71
72 // Select GPU device (OpenMP)
73 ngpus = omp_get_num_devices();
74 if (myid_w == 0) std::printf("num of GPUs = %d\n", ngpus);
75 gpuid = (ngpus > 0) ? (myid_w % ngpus) : -1;
76 if (gpuid >= 0) omp_set_default_device(gpuid);
77
78 printf("init myid_w,gpuid=(%i,%i)\n",myid_w,gpuid);
79 // If device-side use of myid_w is needed later, this updates it once.
80// #pragma acc update device(myid_w)
81}
