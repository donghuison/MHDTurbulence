66static void GenerateProblem(hydflux_mod::GridArray<double>& G,hydflux_mod::FieldArray<double>& P,hydflux_mod::FieldArray<double>& U) {
67 // Kelvin-Helmholtz initial condition
68 using namespace resolution_mod;
69 using namespace mpi_config_mod;
70 using namespace hydflux_mod;
71
72 const double pi = std::acos(-1.0);
73
74 // Parameters (same as Fortran)
75 const double rho1 = 1.0;
76 const double rho2 = 1.0;
77 const double dv = 2.0;
78 const double wid = 0.05;
79 const double sig = 0.2;
80 const double B0 = std::sqrt(2.0/3.0);
81 (void)rho1; (void)rho2; (void)B0; // currently symmetric / B0 unused in reference file
82
83 // Random perturbation strength (Fortran rrv)
84 const double rrv = 0.0e-2;
85
86 // Isothermal: choose csiso so that p = rho*csiso^2 matches Fortran p=1 when rho=1
87 //csiso = 1.0;
88 //#pragma omp target update to (csiso)
89
90 chg = 0.0;
91
92 // Base profile (fill including ghost cells to avoid uninitialized values)
93 for (int k = ks-ngh; k <= ke+ngh; ++k)
94 for (int j = js-ngh; j <= je+ngh; ++j)
95 for (int i = is-ngh; i <= ie+ngh; ++i) {
96 const double x = G.x1b(i);
97 const double y = G.x2b(j);
98
99 const double tanh_p = std::tanh((y + 0.5)/wid);
100 const double tanh_m = std::tanh((y - 0.5)/wid);
101
102 P(nden,k,j,i) = 1.0;
103 P(npre,k,j,i) = 1.0;
104
105 P(nve1,k,j,i) = 0.5 * dv * (tanh_p - tanh_m - 1.0);
106 P(nve2,k,j,i) = 1.0e-2 * std::sin(2.0*pi*x) *
107 ( std::exp(- (y + 0.5)*(y + 0.5)/(sig*sig))
108 + std::exp(- (y - 0.5)*(y - 0.5)/(sig*sig)) );
109 P(nve3,k,j,i) = 0.0;
110
111 P(nbm1,k,j,i) = 0.0;
112 P(nbm2,k,j,i) = 0.0;
113 P(nbm3,k,j,i) = 0.0;
114 P(nbps,k,j,i) = 0.0;
115
116 // specific internal energy is not used in the isothermal closure, but keep it consistent
117 P(nene,k,j,i) = P(npre,k,j,i) /(gam-1.0);
118 P(ncsp,k,j,i) = sqrt(gam*P(npre,k,j,i)/P(nden,k,j,i));
119
120 //P(ncsp,k,j,i) = csiso;
121 // Composition (Fortran: Xcomp(1)=0.5*(tanh((y+0.5)/wid)-tanh((y-0.5)/wid)))
122 for (int n=0; n<ncomp; ++n) {
123 (void)n; // for future multi-comp extension
124 P(nst, k,j,i) = 0.5 * (tanh_p - tanh_m);
125 }
126 }
127
128 // Random perturbation on v1 (Fortran: one random number per (j,k), applied to all i)
129 // Fortran uses random_seed with seed(1)=1, seed(2)=1 + myid_w*in*jn*kn.
130 // Here we emulate with a deterministic per-rank seed.
131 std::mt19937_64 rng(static_cast<unsigned long long>(1 + myid_w * 1000003ULL));
132 std::uniform_real_distribution<double> uni(0.0, 1.0);
133
134 for (int k = ks; k <= ke; k++)
135 for (int j = js; j <= je; j++) {
136 // const double r = uni(rng);
137 // const double dv_rand = dv * rrv * (r - 0.5);
138 for (int i = is; i <= ie; ++i) {
139 //P(nve1,k,j,i) += dv_rand;
140 }
141 }
142
143 namespace res = resolution_mod;
144 for (int k = ks; k <= ke; k++)
145 for (int j = js; j <= je; j++) {
146 const double y = G.x2b(j);
147 const double z = G.x3b(k);
148 const double dv_harm = dv * rrv * std::cos(6*2*pi*(y - res::x2min)/(res::x2max - res::x2min))
149 * std::cos(6*2*pi*(z - res::x3min)/(res::x3max - res::x3min));
150 for (int i = is; i <= ie; i++) {
151 P(nve1,k,j,i) += dv_harm;
152 }
153 }
154
155
156 // Build conserved variables U from primitives P
157 for (int k = ks-ngh; k <= ke+ngh; ++k)
158 for (int j = js-ngh; j <= je+ngh; ++j)
159 for (int i = is-ngh; i <= ie+ngh; ++i) {
160 const double rho = P(nden,k,j,i);
161 U(mden,k,j,i) = rho;
162 U(mrv1,k,j,i) = rho * P(nve1,k,j,i);
163 U(mrv2,k,j,i) = rho * P(nve2,k,j,i);
164 U(mrv3,k,j,i) = rho * P(nve3,k,j,i);
165
166 const double ekin = 0.5 * rho * ( P(nve1,k,j,i)*P(nve1,k,j,i)
167 + P(nve2,k,j,i)*P(nve2,k,j,i)
168 + P(nve3,k,j,i)*P(nve3,k,j,i) );
169 const double emag = 0.5 * ( P(nbm1,k,j,i)*P(nbm1,k,j,i)
170 + P(nbm2,k,j,i)*P(nbm2,k,j,i)
171 + P(nbm3,k,j,i)*P(nbm3,k,j,i) );
172
173 U(meto,k,j,i) = rho * P(nene,k,j,i) + ekin + emag;
174
175 U(mbm1,k,j,i) = P(nbm1,k,j,i);
176 U(mbm2,k,j,i) = P(nbm2,k,j,i);
177 U(mbm3,k,j,i) = P(nbm3,k,j,i);
178 U(mbps,k,j,i) = P(nbps,k,j,i);
179
180 for (int n=0; n<ncomp; ++n) {
181 U(mst+n,k,j,i) = rho * P(nst+n,k,j,i);
182 }
183 }
184}
